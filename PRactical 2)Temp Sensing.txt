const int temperaturepin=A0;
const int ledpin=9;
const float threshold =24;

void setup(){
  
  Serial.begin(9600);
  pinMode(ledpin,OUTPUT);
  Serial.print("Alert thereshold set at :");
  Serial.print(threshold);
  Serial.println('C');
}

void loop(){
  int analog_value=analogRead(temperaturepin);
  float voltage=analog_value* (5.0/1024.0);
  float temperature =(voltage -0.5 )*100;
  
  Serial.print("Current temp:");
  Serial.print(temperature);
  Serial.println("C");
  
  if(temperature > threshold){
    digitalWrite(ledpin,HIGH);
    Serial.println("over temp alert");
    Serial.println(" ");
    
  }
  else{
    digitalWrite(ledpin,LOW);
    Serial.println("status : environment ok");
    Serial.println(" ");
  }

  delay(2000);
}
